select SMS_R_SYSTEM.ResourceID,SMS_R_SYSTEM.ResourceType,SMS_R_SYSTEM.Name,SMS_R_SYSTEM.SMSUniqueIdentifier,SMS_R_SYSTEM.ResourceDomainORWorkgroup,SMS_R_SYSTEM.Client from SMS_R_System    JOIN SMS_UpdateComplianceStatus      ON SMS_R_System.ResourceID = SMS_UpdateComplianceStatus.MachineID      AND SMS_UpdateComplianceStatus.Status = 2    JOIN SMS_SoftwareUpdate      ON SMS_UpdateComplianceStatus.CI_ID = SMS_SoftwareUpdate.CI_ID      AND SMS_SoftwareUpdate.ArticleID = 4012212
